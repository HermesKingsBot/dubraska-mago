# Next.js + Prisma Backend

Build backend APIs in Next.js projects using Prisma ORM. Covers REST API routes, database schema design, authentication, validation, and data persistence.

## Prisma Schema Patterns

### SQLite (Development)

```prisma
generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "sqlite"
  url      = "file:./dev.db"
}

model Product {
  id          String   @id @default(cuid())
  name        String
  slug        String   @unique
  price       Float
  stock       Int      @default(0)
  categoryId  String
  category    Category @relation(fields: [categoryId], references: [id])
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt

  @@index([categoryId])
  @@index([slug])
}
```

### PostgreSQL (Production)

```prisma
datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}
```

### SQLite → PostgreSQL Migration (Full Workflow)

When migrating from SQLite to PostgreSQL (e.g., Neon):

```bash
# 1. Update schema.prisma
# Change: provider = "sqlite" → provider = "postgresql"

# 2. Remove old SQLite migrations and database
rm -rf prisma/migrations
rm -f prisma/dev.db

# 3. Create migration_lock.toml
mkdir -p prisma/migrations/000001_init
cat > prisma/migrations/migration_lock.toml << 'EOF'
# Please do not edit this file manually
# It should be added in your version-control system (e.g., Git)
provider = "postgresql"
EOF

# 4. Generate migration SQL from schema
npx prisma migrate diff --from-empty --to-schema-datamodel prisma/schema.prisma --script > prisma/migrations/000001_init/migration.sql

# 5. Apply migration to PostgreSQL
DATABASE_URL="postgresql://..." npx prisma migrate deploy

# 6. Generate Prisma Client
npx prisma generate

# 7. Configure Vercel environment variable
source .env.vercel
DATABASE_URL=$(grep "^DATABASE_URL=" .env | cut -d'"' -f2)
echo "$DATABASE_URL" | npx vercel --token "$VERCEL_TOKEN" env add DATABASE_URL production

# 8. Add postinstall script to package.json
# "postinstall": "prisma generate"

# 9. Deploy
npx vercel --token "$VERCEL_TOKEN" deploy --prod --force
```

**Note:** SQLite doesn't support arrays, enums, or JSON natively. Use:
- Arrays: Store as JSON string (`gallery: String @default("[]")`), parse on read
- Enums: Store as String, validate with Zod
- JSON: Store as String, parse with `JSON.parse()`

**Neon Postgres specifics:**
- Connection string format: `postgresql://user:pass@host:5432/db?sslmode=require`
- No `pgbouncer=true` needed for basic setup
- Serverless — scales to zero when idle
- Works with Prisma 6.x + Next.js 16

## API Route Patterns

### List + Create (GET + POST)

```typescript
// src/app/api/products/route.ts
import { NextRequest, NextResponse } from "next/server"
import db from "@/lib/db"
import { successResponse, handleApiError } from "@/lib/api"
import { createProductSchema } from "@/lib/schemas"

export async function GET(request: NextRequest): Promise<NextResponse> {
  try {
    const { searchParams } = new URL(request.url)
    const category = searchParams.get("category") || undefined
    const page = parseInt(searchParams.get("page") || "1")
    const limit = parseInt(searchParams.get("limit") || "20")

    const where: Record<string, unknown> = {}
    if (category) where.categoryId = category

    const [items, total] = await Promise.all([
      db.product.findMany({
        where,
        include: { category: true },
        orderBy: { createdAt: "desc" },
        skip: (page - 1) * limit,
        take: limit,
      }),
      db.product.count({ where }),
    ])

    return successResponse({ items, total, page, totalPages: Math.ceil(total / limit) })
  } catch (error) {
    return handleApiError(error)
  }
}

export async function POST(request: NextRequest): Promise<NextResponse> {
  try {
    const body = await request.json()
    const data = createProductSchema.parse(body)
    const product = await db.product.create({ data })
    return successResponse(product, 201)
  } catch (error) {
    return handleApiError(error)
  }
}
```

### Single Resource (GET + PATCH + DELETE)

```typescript
// src/app/api/products/[id]/route.ts
import { NextRequest, NextResponse } from "next/server"
import db from "@/lib/db"
import { successResponse, errorResponse, handleApiError } from "@/lib/api"
import { updateProductSchema } from "@/lib/schemas"

export async function GET(
  _request: NextRequest,
  { params }: { params: Promise<{ id: string }> }
): Promise<NextResponse> {
  try {
    const { id } = await params
    const product = await db.product.findUnique({ where: { id }, include: { category: true } })
    if (!product) return errorResponse("Not found", 404)
    return successResponse(product)
  } catch (error) {
    return handleApiError(error)
  }
}

export async function PATCH(
  request: NextRequest,
  { params }: { params: Promise<{ id: string }> }
): Promise<NextResponse> {
  try {
    const { id } = await params
    const body = await request.json()
    const data = updateProductSchema.parse(body)
    const product = await db.product.update({ where: { id }, data })
    return successResponse(product)
  } catch (error) {
    return handleApiError(error)
  }
}

export async function DELETE(
  _request: NextRequest,
  { params }: { params: Promise<{ id: string }> }
): Promise<NextResponse> {
  try {
    const { id } = await params
    await db.product.delete({ where: { id } })
    return successResponse({ deleted: true })
  } catch (error) {
    return handleApiError(error)
  }
}
```

## API Response Helpers

```typescript
// src/lib/api.ts
import { NextResponse } from "next/server"
import { ZodError } from "zod"

export function successResponse<T>(data: T, status = 200): NextResponse {
  return NextResponse.json({ success: true, data }, { status })
}

export function errorResponse(message: string, status = 400): NextResponse {
  return NextResponse.json({ success: false, error: message }, { status })
}

export function handleApiError(error: unknown): NextResponse {
  if (error instanceof ZodError) {
    const messages = error.errors.map((e) => `${e.path.join(".")}: ${e.message}`).join(", ")
    return errorResponse(messages, 422)
  }
  console.error(error)
  return errorResponse("Internal server error", 500)
}
```

## Validation Schemas (Zod)

```typescript

import { z } from "zod"

export const createProductSchema = z.object({
  name: z.string().min(1).max(200),
  slug: z.string().regex(/^[a-z0-9-]+$/, "Slug must be lowercase alphanumeric"),
  price: z.number().positive(),
  stock: z.number().int().min(0).optional(),
  categoryId: z.string().min(1),
  color: z.enum(["dorado", "plateado", "rose", "negro"]),
})

export const updateProductSchema = createProductSchema.partial()

export const loginSchema = z.object({
  email: z.string().email(),
  password: z.string().min(1),
})
```

## JWT Authentication

```typescript
// src/lib/auth.ts
import { NextRequest } from "next/server"
import jwt from "jsonwebtoken"

const JWT_SECRET = process.env.JWT_SECRET || "fallback-secret"

interface JwtPayload {
  userId: string
  email: string
  role: string
}

export function verifyToken(request: NextRequest): JwtPayload | null {
  const authHeader = request.headers.get("authorization")
  if (!authHeader?.startsWith("Bearer ")) return null
  try {
    return jwt.verify(authHeader.split(" ")[1], JWT_SECRET) as JwtPayload
  } catch {
    return null
  }
}

export function requireAuth(request: NextRequest): JwtPayload {
  const user = verifyToken(request)
  if (!user) throw new Error("Unauthorized")
  return user
}
```

```typescript
// src/app/api/auth/login/route.ts
import { NextRequest, NextResponse } from "next/server"
import bcrypt from "bcryptjs"
import jwt from "jsonwebtoken"
import db from "@/lib/db"
import { successResponse, errorResponse, handleApiError } from "@/lib/api"
import { loginSchema } from "@/lib/schemas"

const JWT_SECRET = process.env.JWT_SECRET || "fallback-secret"

export async function POST(request: NextRequest): Promise<NextResponse> {
  try {
    const body = await request.json()
    const { email, password } = loginSchema.parse(body)

    const user = await db.adminUser.findUnique({ where: { email } })
    if (!user || !user.active) return errorResponse("Invalid credentials", 401)

    const isValid = await bcrypt.compare(password, user.password)
    if (!isValid) return errorResponse("Invalid credentials", 401)

    const token = jwt.sign(
      { userId: user.id, email: user.email, role: user.role },
      JWT_SECRET,
      { expiresIn: "7d" }
    )

    return successResponse({ token, user: { id: user.id, email: user.email, name: user.name } })
  } catch (error) {
    return handleApiError(error)
  }
}
```

**Key patterns:**
- Use `Promise.all()` for parallel creation of independent records
- Delete in reverse FK dependency order (children before parents)
- Use `import * as bcrypt from "bcryptjs"` (not default import — TS1259 error)
- Store JSON arrays as strings: `JSON.stringify([...])`
- Include realistic test data: categories, products, testimonials, users, settings, social links, reviews, inventory movements
- Log summary at end: counts of created records

```typescript
// src/app/api/seed/route.ts
import { NextResponse } from "next/server"
import db from "@/lib/db"
import { successResponse, handleApiError } from "@/lib/api"
import products from "../../../../data/products.json"

export async function POST(): Promise<NextResponse> {
  try {
    await db.product.deleteMany()
    await db.category.deleteMany()

    const uniqueCategories = [...new Set(products.map((p: { category: string }) => p.category))]
    const categoryMap: Record<string, string> = {}

    for (const catName of uniqueCategories) {
      const cat = await db.category.create({
        data: { name: catName, slug: catName, active: true },
      })
      categoryMap[catName] = cat.id
    }

    for (const product of products as Record<string, unknown>[]) {
      await db.product.create({
        data: {
          name: product.name as string,
          slug: product.slug as string,
          price: product.price as number,
          stock: 15,
          categoryId: categoryMap[product.category as string],
          color: product.color as string,
          image: product.image as string,
          gallery: "[]",
        },
      })
    }

    return successResponse({ seeded: true, products: products.length, categories: uniqueCategories.length })
  } catch (error) {
    return handleApiError(error)
  }
}
```

## Inventory Transaction Pattern

```typescript
// src/app/api/inventory/adjust/route.ts
import { NextRequest, NextResponse } from "next/server"
import db from "@/lib/db"
import { successResponse, handleApiError } from "@/lib/api"
import { inventoryAdjustmentSchema } from "@/lib/schemas"

export async function POST(request: NextRequest): Promise<NextResponse> {
  try {
    const body = await request.json()
    const data = inventoryAdjustmentSchema.parse(body)

    const movement = await db.$transaction(async (tx) => {
      await tx.inventoryMovement.create({
        data: {
          productId: data.productId,
          type: data.type,
          quantity: data.quantity,
          reason: data.reason,
        },
      })

      const product = await tx.product.findUnique({ where: { id: data.productId } })
      if (!product) throw new Error("Product not found")

      let newStock = product.stock
      if (data.type === "STOCK_IN") newStock += data.quantity
      else if (data.type === "STOCK_OUT") newStock = Math.max(0, newStock - data.quantity)
      else newStock = data.quantity

      await tx.product.update({ where: { id: data.productId }, data: { stock: newStock } })
      return { id: product.id, stock: newStock }
    })

    return successResponse(movement, 201)
  } catch (error) {
    return handleApiError(error)
  }
}
```

## Pitfalls

1. **SQLite arrays**: Store as JSON string, parse with `JSON.parse()` on read
2. **SQLite enums**: Use String + Zod validation, migrate to native enum when switching to PostgreSQL
3. **params is Promise**: Always `await params` in Next.js 15+
4. **Prisma singleton**: Use `globalThis` pattern to avoid multiple instances in dev
5. **Migration conflicts**: If schema changes drastically, `prisma migrate reset` then re-seed
6. **JSON fields in SQLite**: Prisma returns String for JSON columns — always parse manually
7. **Auth in Server Actions**: Always validate auth inside the action, never trust client
8. **Transaction errors**: Wrap related writes in `prisma.$transaction()` for atomicity
9. **Optional field cascade**: When moving fields from a base model to a child model (e.g., Product → ProductVariant), make base fields nullable (`price Float?`, `color String?`). After migration, ALL code referencing `product.price`, `product.color`, etc. needs null-safety: `(product.price ?? 0)`, `product.color ?? ""`, `product.color ? ... : "—"`. Grep for `.<field>.` across the codebase to find all affected files.
10. **Prisma 6 with Next.js 16**: Always use Prisma 6.x (not 7.x) with Next.js 16/Turbopack. Schema uses `env("DATABASE_URL")` in datasource block.
